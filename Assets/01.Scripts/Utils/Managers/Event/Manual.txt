How to use:
	EventManager.TriggerEvent("gameOver", null);

Use with param:
	EventManager.TriggerEvent("gamePause", new Dictionary<string, object> { { "pause", true } });

	EventManager.TriggerEvent("addReward", 
  		new Dictionary<string, object> {
  		  { "name", "candy" },
   		 { "amount", 5 } 
  		});
